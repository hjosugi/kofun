  let answer = (

  )
  let values = [
    answer,
    4,
  ]
  let total = values
  |> map(fn(value: Int) => value * 2)
  |> sum()
  print(total)
}

fn classify(value: Int) -> Text {
  return if value < 0 {
    "negative"
  } else if value == 0 {
    "zero"
  } else {
    "positive"
  }
}

fn main() {
  let mut value = 0
  while value < 10 {
    value = value + 1
  }
  for item in 0 .. value {
    if item == 4 {
      continue
    }
  }
}
